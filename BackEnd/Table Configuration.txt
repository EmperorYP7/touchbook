YOU NEED TO CREATE BELOW TWO MYSQL TABLE IN YOUR BACKEND FOR THE APP TO WORK

CREATE TABLE `2734291_data`.`touch_auth` ( `uid` BIGINT NOT NULL AUTO_INCREMENT , 
`userid` VARCHAR(50) CHARACTER SET latin1 COLLATE latin1_swedish_ci NOT NULL , 
`password` VARCHAR(50) CHARACTER SET latin1 COLLATE latin1_swedish_ci NOT NULL , 
UNIQUE `uid` (`uid`)) ENGINE = MyISAM;

CREATE TABLE `2734291_data`.`touch_data` ( `uid` BIGINT NOT NULL AUTO_INCREMENT , 
`name` VARCHAR(50) CHARACTER SET latin1 COLLATE latin1_spanish_ci NOT NULL , 
`phone` VARCHAR(50) CHARACTER SET latin1 COLLATE latin1_swedish_ci NOT NULL , 
`batch` VARCHAR(50) CHARACTER SET latin1 COLLATE latin1_swedish_ci NOT NULL , 
`company` VARCHAR(50) CHARACTER SET latin1 COLLATE latin1_swedish_ci NOT NULL , 
UNIQUE `uid` (`uid`)) ENGINE = MyISAM;
